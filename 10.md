# Matplotlib
```
import matplotlib.pyplot as plt
x=[1,2,3]
y=[-2,-4,-8]
y1=[1,3,7]
plt.plot(x,y) # send the data to matplot
plt.plot(x,y1)
plt.show()

import numpy as np # we can also use the numpy data and methods.
X_1=range(100)
Y_1=[np.cos(i) for i in X_1]

X_2=range(100)
Y_2=[np.sin(i) for i in X_1]
plt.plot(X_1,Y_1)
plt.plot(X_2,Y_2)
```
